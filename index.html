/**
* Build a pool of questions from selected chapters, balancing across chapters.
* Returns questions with metadata: {chapter, difficulty, topic, text, answer}
*/
function buildPool(chapters, diff){
// call each generator multiple times for variety
const byChapter={};
chapters.forEach(ch=>{
const gen=G[ch];if(!gen)return;
const levelFns=gen[diff]||gen['easy']||[];
const fallbackFns=(diff!=='easy'&&gen['easy'])||[];
const allFns=[...levelFns,...(levelFns.length<5?fallbackFns:[])];
byChapter[ch]=[];
const shuffled=shuffle(allFns);
shuffled.forEach(fn=>{
for(let i=0;i<3;i++){
try{
const q=fn();
byChapter[ch].push({chapter:ch,difficulty:diff,topic:q.topic||'General',text:q.text,answer:q.ans
wer});
}catch(e){}
}
});
// deduplicate within chapter
const seen=new Set();
byChapter[ch]=byChapter[ch].filter(q=>{if(seen.has(q.text))return
false;seen.add(q.text);return true});
});
return byChapter;
}
/**
* Pick n questions, ensuring balanced chapter distribution.
*/
function pickBalanced(byChapter,count){
const chapters=Object.keys(byChapter);
if(chapters.length===0)return[];
const perChapter=Math.ceil(count/chapters.length);
const result=[];
const seen=new Set();
// first pass: pick up to perChapter from each chapter
chapters.forEach(ch=>{
const pool=shuffle(byChapter[ch]);
let taken=0;
for(const q of pool){
if(taken>=perChapter||result.length>=count)break;
if(!seen.has(q.text)){seen.add(q.text);result.push(q);taken++;}
}
});
// second pass: fill remainder from any chapter
if(result.length<count){
const allRemaining=shuffle(chapters.flatMap(ch=>byChapter[ch]));
for(const q of allRemaining){
if(result.length>=count)break;
if(!seen.has(q.text)){seen.add(q.text);result.push(q);}
}
}
return shuffle(result).slice(0,count);
}
function buildPaper(chapters,diff){
const pool=buildPool(chapters,diff);
const easyPool=buildPool(chapters,'easy');
const modPool=buildPool(chapters,'moderate');
const hardPool=buildPool(chapters,'hard');
return{
chapters,difficulty:diff,
sections:[
{label:'Section A',instruction:'Answer all questions.',marks_each:1,total_marks:10,
questions:pickBalanced(diff==='hard'?modPool:easyPool,10)},
{label:'Section B',instruction:'Attempt any five questions.',marks_each:3,total_marks:15,
questions:pickBalanced(diff==='easy'?easyPool:modPool,6)},
{label:'Section C',instruction:'Attempt any four questions.',marks_each:4,total_marks:16,
questions:pickBalanced(diff==='easy'?modPool:hardPool,5)},
{label:'Section D',instruction:'Attempt any two questions.',marks_each:5,total_marks:10,
questions:pickBalanced(hardPool,3)},
]
};
}
